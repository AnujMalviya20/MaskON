const uploadArea = document.getElementById('uploadArea'); const fileInput = document.getElementById('fileInput'); const previewImage = document.getElementById('previewImage'); const previewSection = document.getElementById('previewSection'); const detectBtn = document.getElementById('detectBtn'); const statusText = document.getElementById('statusText'); const faceCountText = document.getElementById('faceCountText'); const confidenceText = document.getElementById('confidenceText'); const masksText = document.getElementById('masksText'); const noMasksText = document.getElementById('noMasksText'); const resultsList = document.getElementById('resultsList'); const resultsSection = document.getElementById('resultsSection'); let selectedFile = null; // Upload area click uploadArea.addEventListener('click', () => fileInput.click()); // File input change fileInput.addEventListener('change', (e) => { const file = e.target.files[0]; if (file && file.type.startsWith('image/')) { selectedFile = file; const reader = new FileReader(); reader.onload = (e) => { previewImage.src = e.target.result; previewSection.style.display = 'block'; resultsSection.style.display = 'none'; statusText.textContent = 'Ready'; statusText.style.color = '#06b6d4'; }; reader.readAsDataURL(file); } }); // Drag and drop uploadArea.addEventListener('dragover', (e) => { e.preventDefault(); uploadArea.classList.add('dragover'); }); uploadArea.addEventListener('dragleave', () => { uploadArea.classList.remove('dragover'); }); uploadArea.addEventListener('drop', (e) => { e.preventDefault(); uploadArea.classList.remove('dragover'); const files = e.dataTransfer.files; if (files.length > 0) { fileInput.files = files; fileInput.dispatchEvent(new Event('change', { bubbles: true })); } }); // Detect button detectBtn.addEventListener('click', async () => { if (!selectedFile) return; try { statusText.textContent = 'Analyzing...'; statusText.style.color = '#f59e0b'; detectBtn.disabled = true; const formData = new FormData(); formData.append('file', selectedFile); const response = await fetch((window.APP_CONFIG?.api_base || 'http://127.0.0.1:5001') + '/api/upload', { method: 'POST', body: formData }); if (!response.ok) throw new Error('Upload failed'); const result = await response.json(); if (result.success) { faceCountText.textContent = result.count; confidenceText.textContent = result.confidence + '%'; masksText.textContent = result.masks_detected; noMasksText.textContent = result.no_masks_detected; statusText.textContent = result.count > 0 ? 'Complete' : 'No Faces'; statusText.style.color = result.count > 0 ? '#10b981' : '#94a3b8'; resultsSection.style.display = 'block'; updateResults(result.detections); } else { statusText.textContent = 'Error'; statusText.style.color = '#ef4444'; } } catch (error) { console.error('Error:', error); statusText.textContent = 'Error'; statusText.style.color = '#ef4444'; alert('Error: ' + error.message); } finally { detectBtn.disabled = false; } }); // Update results display function updateResults(detections) { if (detections.length === 0) { resultsList.innerHTML = '
No faces detected
'; return; } resultsList.innerHTML = detections.map((det, idx) => { const isMask = det.label === 'MASK DETECTED'; const badgeClass = isMask ? 'badge-mask' : 'badge-no-mask'; const icon = isMask ? '✓' : '✗'; return `
${icon}
Face ${idx + 1}: ${det.label}
${det.confidence}%
`; }).join(''); }
